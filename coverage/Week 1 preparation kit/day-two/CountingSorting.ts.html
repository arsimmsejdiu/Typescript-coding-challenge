export function comparisonSorting(arr: number[]): number[] {
  const count: number[] = new Array(100).fill(0); // Since we need to count numbers from 0 to 99
 
  // Count frequencies of each element in arr
  for (let num of arr) {
    count[num]++;
  }
 
  return count;
}
